How Rust handles this

  • Rust doesn't have the concept of throwing errors
  • No null or undefined
  • Rust has two enums for this

Option


              pub enum Option<T> {
                None,
                Some(T)
              }

Example


              fn divide(dividend: i32, divisor: i32) -> Option <i32> {
                if divisor == 0 {
                    // Failure is represented as the `None` variant
                    None
                } else {
                    // Result is wrapped in a `Some` variant
                    Some(dividend / divisor)
                }
              }

Generics


              function reverse<T>(items: T[]): T[] {
                let reversedArray = [];
                for (let i = items.length - 1; i >= 0; i--) {
                    reversedArray.push(items[i]);
                }
                return reversedArray;
              }

Discriminated Unions


              type Square = {
                kind: 'square';
                size: number;
              }
              
              type Rectangle = {
                kind: 'rectangle';
                width: number;
                height: number;
              }
              
              type Shape = Square | Rectangle;
            

              function area(s: Shape) {
                if (s.kind === 'square') {
                  // Now TypeScript knows that `s` must be a square
                  return s.size * s.size;
                }
                else {
                  // Not a square? 
                  // So TypeScript figures out that it's a Rectangle
                  return s.width * s.height;
                }
              }

Type Predicates


              function isString(x: unknown): x is string {
                return typeof x === 'string';
              }

              if (isString(x)) {
                // Here x is inferred a string
                return x.split('').reverse().join('');
              }

Result


              pub enum Result<T, E> {
                Ok(T),
                Err(E)
              }

Example


              fn divide(x: i32, y: i32) -> Result<i32, DivisionError> {
                if y == 0 {
                    return Err(DivisionError::DivideByZero);
                }
                Ok(x / y)
              }

Rust helper methods (non-exhaustive)

unwrap


              let x = Some("air");
              assert_eq!(x.unwrap(), "air");

              // this will fail (rust panic)
              let x: Option<&str> = None;
              assert_eq!(x.unwrap(), "air");

unwrap_or


              assert_eq!(Some("car").unwrap_or("bike"), "car");
              assert_eq!(None.unwrap_or("bike"), "bike");
